\begin{frame}
    \frametitle{From Ortholattices to Orthologic}

    % notably, if you restrict classical logic to have atmost one formula on the
    % right hand side, you get intuitionistic logic

    Intuitionistic logic --- at most one formula on the right-hand side of the sequent

    \begin{gather*}
        \gamma_1, \gamma_2, \ldots \vdash \psi
    \end{gather*}

    \pause

    % analogously, orthologic, embodying the laws of ortholattices, is
    % characterized by having atmost two formulas in the sequent at any step

    Orthologic --- at most two formulas in the entire sequent

    \begin{gather*}
        \phi \vdash \psi \\
        \phi, \psi \vdash \\
        \vdash \phi, \psi
    \end{gather*}

    \pause

    % as it's more convenient for orthologic, we write sequents as a set of
    % (atmost two) formulas with left and right annotations

    For convenience, written with left/right annotations, e.g. \(\phi^L, \psi^R\)

\end{frame}

\begin{frame}
    \frametitle{Interpolation}

    % once we have these proofs, we are interested in interpolation

    % can we find interpolating formulas or "local proofs"

    % interpolation is the question of whether, given say two formulas A and B,
    % and a proof that A implies B, can we find a formula I such that A implies
    % I, and I implies B, while I contains only the free variables common to A
    % and B.

    % in essence, finding the relevant facts in the proof which link the
    % hypothesis and the conclusion

    Given that
    \begin{gather*}
        A_{\bar{x}, \bar{y}} \implies B_{\bar{y}, \bar{z}}
    \end{gather*}

    \pause
    find \(I_{\bar{y}}\) such that
    \(A_{\bar{x}, \bar{y}} \implies I_{\bar{y}}\), and
    \(I_{\bar{y}} \implies B_{\bar{y}, \bar{z}}\).

    \pause
    \begin{itemize}
        \item Focus search to relevant facts % less explosion
        \item Better counterexamples
        \item Abstraction generalization
        \footnote<3->{
            \Citeauthor{mcmillan2018interpolation}. \Citetitle{mcmillan2018interpolation}. In: Handbook of Model Checking
        }
    \end{itemize}

\begin{frame}
    \frametitle{Interpolation in Boolean Algebras}

    % note that if we allow quantifiers, we get some interpolants for free
    % by simply quantifying away the varaibles that are not common to A and B

    % Boolean algebras, in particular, admit quantifier elimination, which means
    % that for these formulas containing quantifiers (which are not in the
    % language of BA that we need), we can compute equivalent quantifier-free
    % formulas, giving us the interpolants we wanted

    \begin{gather*}
        A_{\bar{x}, \bar{y}} \implies B_{\bar{y}, \bar{z}}
    \end{gather*}

    \pause
    With quantifiers, we get some interpolants for free:

    \begin{itemize}
        \item \(I_1 : \exists \bar{x}. A_{\bar{x}, \bar{y}}\)
        \item \(I_2 : \forall \bar{z}. B_{\bar{y}, \bar{z}}\)
    \end{itemize}

    % we can prove with a little effort that these are indeed interpolants

    \pause
    By quantifier elimination for Boolean algebras, the quantifier-free formulas
    \(QE(I_1)\) and \(QE(I_2)\) can be computed. 

    \pause
    Better: we get least and most general interpolants. \footnote{\Citeauthor{dsilva2010interpolant}. \emph{\Citetitle{dsilva2010interpolant}}. In: VMCAI 2010}
    % also allows you to do some symbolic state computation
\end{frame}

\begin{frame}
    \frametitle{Quantified Orthologic}

    % we show that this new proof system is both sound and complete for
    % ortholattices

    \begin{theorem}[Soundness]
        For every sequent \(S\), if \(\vdash S\) then \(\models S\).
    \end{theorem}

    \begin{theorem}[Completeness]
        For every sequent \(S\), if \(\models S\) then \(\vdash S\).
    \end{theorem}

    \pause

    Inequalities correspond directly to sequents:
    \begin{gather*}
        \gamma \leq \delta \iff \vdash (\gamma^L, \delta^R)
    \end{gather*}

\begin{frame}
    \frametitle{Refutation-based Interpolation}

    % so we consider two different notions of interpolation, with the first
    % based on refutation proofs.

    Given two sequents \(\gamma\) and \(\delta\), and a proof of contradiction assuming
    \(\gamma\) and \(\delta\), find an interpolant sequent \(I\), such that

    \begin{itemize}[<+->]
        \item \(I\) can be deduced from \(\gamma\) alone
        \item \(I\) and \(\delta\) are inconsistent when assumed together
        \item \(FV(I) \subset FV(\gamma) \cap FV(\delta)\)
    \end{itemize}

    % and we show that orthologic does not admit refutation-based interpolation
    % also by counterexample. 

    % there is an OL proof of contradiction assuming \gamma and \delta, which
    % you can find in the paper 
    
    % Same as quantifier elimination, the interpolant for these two formulas
    % must only contain y as a free variable, and we show that no such formula
    % can be an interpolant for this example.

    \pause
    \begin{theorem}
        Orthologic does not admit refutation-based interpolation.
    \end{theorem}

    \pause

    Counterexample: 
    \begin{gather*}
        \gamma: (z \lor \neg y) \land (\neg z \lor \neg y)^R \\
        \delta: (x \land \neg y) \lor (\neg x \land \neg y)^R
    \end{gather*}
    \pause
    No sequent \(I\) containing only \(y\) can be a refutation-based interpolant
    for this example.

\end{frame}

% third result: interpolation 2 does exist, linear in proof size
\begin{frame}
    \frametitle{Implicational Interpolation for Ortholattices}

    % finally, we consider a notion of interpolation that is more natural for
    % lattices, based on implications, or lattice inequalities, i.e.
    % given two formulas \gamma and \delta, such that \gamma < \delta 

    Given two ortholattice formulas \(\gamma\) and \(\delta\), such that
    \(\gamma \leq \delta\) wrt laws of OL, an implicational interpolant is a
    formula \(I\) such that

    \begin{itemize}
        \item \(\gamma \leq I\)
        \item \(I \leq \delta\)
        \item \(FV(I) \subset FV(\gamma) \cap FV(\delta)\)
    \end{itemize}

\end{frame}

\begin{frame}
    \frametitle{Implicational Interpolation for Ortholattices}

    Given a provable sequent \(\Gamma, \Delta\), an implicational interpolant is
    a formula \(I\) such that

    \begin{itemize}
        \item \(\Gamma, I^R\)
        \item \(I^L, \Delta\)
        \item \(FV(I) \subset FV(\Gamma) \cap FV(\Delta)\)
    \end{itemize}

    \pause

    \begin{theorem}
        Orthologic admits implicational interpolation.
    \end{theorem}

    \pause
    \begin{itemize}
        \item linear-time in size of the proof
        \item thus quadratic-time in size of the sequent
    \end{itemize}
    
\end{frame}
